    Implement as an NSObject subclass (REQUIRED so the IOBluetooth target/selector dispatch can find the @objc methods — Pitfall 5) and `@MainActor`:

    ```swift
    import IOBluetooth
    import AppKit

    @MainActor
    final class BluetoothMonitor: NSObject {
        // Connect class-notification: "not necessary to retain" but we hold it to unregister in stop().
        private nonisolated(unsafe) var connectToken: IOBluetoothUserNotification?
        // Per-device disconnect tokens — MUST be kept alive until they fire (Pitfall 4). Keyed by addressString.
        private nonisolated(unsafe) var disconnectTokens: [String: IOBluetoothUserNotification] = [:]
        // The raw event the controller maps via deviceActivity(from: DeviceReading).
        private let onEvent: (_ name: String?, _ classMajor: UInt32, _ address: String?, _ connected: Bool) -> Void

        init(onEvent: @escaping (String?, UInt32, String?, Bool) -> Void) {
            self.onEvent = onEvent
            super.init()
        }

        func start() {
            // Register the app-wide CONNECT class-notification on MAIN so callbacks arrive on main.
            connectToken = IOBluetoothDevice.register(forConnectNotifications: self,
                                                      selector: #selector(deviceConnected(_:device:)))
        }

        @objc private func deviceConnected(_ note: IOBluetoothUserNotification, device: IOBluetoothDevice) {
            let addr = device.addressString
            // Pitfall 4: register + RETAIN a per-device disconnect token (only once per address).
            if let addr, disconnectTokens[addr] == nil {
                disconnectTokens[addr] = device.register(forDisconnectNotification: self,
                                                         selector: #selector(deviceDisconnected(_:device:)))
            }
            // Pitfall 5: defensively hop to main before the closure touches @Published/AppKit.
            DispatchQueue.main.async { self.onEvent(device.name, device.deviceClassMajor, addr, true) }
        }

        @objc private func deviceDisconnected(_ note: IOBluetoothUserNotification, device: IOBluetoothDevice) {
            let addr = device.addressString
            if let addr { disconnectTokens[addr]?.unregister(); disconnectTokens[addr] = nil }
            DispatchQueue.main.async { self.onEvent(device.name, device.deviceClassMajor, addr, false) }
        }

        // nonisolated so the controller's nonisolated deinit can call it (mirrors PowerSourceMonitor.stop()).
        nonisolated func stop() {
            connectToken?.unregister(); connectToken = nil
            disconnectTokens.values.forEach { $0.unregister() }
            disconnectTokens.removeAll()
        }

        deinit {
            // Like PowerSourceMonitor: deinit can't be @MainActor in Swift 5 mode → the @MainActor
            // controller's deinit calls bluetoothMonitor.stop() to unregister all tokens.
        }
    }
    ```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| IOBluetooth callback thread → main | Callbacks may arrive off-main; touching @Published/AppKit off-main corrupts UI |
| notification token lifetime → app lifetime | A token used after its owner is freed crashes; an under-retained disconnect token silently never fires |
| macOS TCC → app | Active BT calls (pairedDevices/scan) trigger an intrusive prompt (Success Criterion 3) |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-05-02 | Tampering / UI corruption | BluetoothMonitor callbacks | mitigate | Every connect/disconnect callback `DispatchQueue.main.async`-hops before invoking `onEvent` (which the controller uses to mutate @Published) — mirrors PowerSourceMonitor's main-hop (Pitfall 5 / PowerSourceMonitor T-03 discipline). |
| T-05-03 | Tampering / crash (use-after-free) | connect + disconnect notification tokens | mitigate | Per-device disconnect tokens retained in a `[String: IOBluetoothUserNotification]` until they fire (Pitfall 4); `nonisolated stop()` `.unregister()`s the connect token AND every stored disconnect token, called from the controller's nonisolated deinit (Plan 03) — mirrors `powerMonitor.stop()` T-03-06. |
| T-05-04 | Information disclosure / privacy (intrusive prompt) | IOBluetooth registration | mitigate | Monitor registers ONLY notifications — NO `pairedDevices()`, NO scanning/inquiry, NO connection opening (the prompt-triggering calls). Usage-description key added ONLY if the Plan-01 spike proved it required, keeping any prompt one-time + expected (Criterion 3). |
</threat_model>
