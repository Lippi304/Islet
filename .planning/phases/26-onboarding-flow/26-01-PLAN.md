From Islet/Notch/IslandResolver.swift (current, unmodified -- lines 1-54):
```
import Foundation

enum IslandPresentation: Equatable {
    case idle
    case charging(ChargingActivity)
    case device(DeviceActivity)
    case nowPlayingWings(NowPlayingPresentation)
    case nowPlayingExpanded(NowPlayingPresentation, healthy: Bool)
    case expandedIdle
}

enum ActiveTransient: Equatable {
    case charging(ChargingActivity)
    case device(DeviceActivity)
}

func resolve(activeTransient: ActiveTransient?,
             nowPlaying: NowPlayingPresentation,
             nowPlayingHealthy: Bool,
             hasPlayedSinceLaunch: Bool,
             isExpanded: Bool) -> IslandPresentation {
    switch activeTransient {
    case .charging(let a): return .charging(a)
    case .device(let d):   return .device(d)
    case nil: break
    }
    if isExpanded {
        if !nowPlayingHealthy { return .nowPlayingExpanded(nowPlaying, healthy: false) }
        if nowPlaying != .none { return .nowPlayingExpanded(nowPlaying, healthy: true) }
        return .expandedIdle
    }
    let ambient = nowPlayingLaunchGate(hasPlayedSinceLaunch: hasPlayedSinceLaunch, nowPlaying: nowPlaying)
    if ambient != .none { return .nowPlayingWings(ambient) }
    return .idle
}
```

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-26-02 | Tampering | `IslandResolver.resolve(...)` onboarding precedence | mitigate | `onboardingStep` checked as the literal first statement of the pure reducer, covered by `testOnboardingOutranksEverything()` -- a forced onboarding session cannot be silently bypassed by a crafted transient/expanded combination, because the check happens before any other branch is reachable |
| T-26-03 | Tampering | `ActivitySettings.onboardingCompletedKey` (plain UserDefaults) | accept | Deliberately not Keychain-backed (this is a UX gate, not a security boundary per CONTEXT.md/RESEARCH.md) -- a tampered value at worst re-shows or permanently skips onboarding once; it can never affect `LicenseState`/trial state, which lives in a completely separate, unmodified store |
