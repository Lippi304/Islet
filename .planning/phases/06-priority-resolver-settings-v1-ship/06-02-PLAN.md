```swift
import IOBluetooth
import AppKit

@MainActor
final class BluetoothMonitor: NSObject {
    private var connectToken: IOBluetoothUserNotification?
    private var disconnectTokens: [String: IOBluetoothUserNotification] = [:]
    private var running = false                            // Pitfall 5: idempotent start()
    private let onReading: (DeviceReading) -> Void

    init(onReading: @escaping (DeviceReading) -> Void) { self.onReading = onReading; super.init() }

    func start() {
        guard !running else { return }                     // Pitfall 5
        running = true
        connectToken = IOBluetoothDevice.register(forConnectNotifications: self,
                                                  selector: #selector(connected(_:device:)))
    }

    @objc private func connected(_ n: IOBluetoothUserNotification, device: IOBluetoothDevice) {
        if let addr = device.addressString, disconnectTokens[addr] == nil {
            disconnectTokens[addr] = device.register(forDisconnectNotification: self,
                                                     selector: #selector(disconnected(_:device:)))  // retain (Pitfall 5)
        }
        emit(device, connected: true)
    }
    @objc private func disconnected(_ n: IOBluetoothUserNotification, device: IOBluetoothDevice) {
        if let addr = device.addressString { disconnectTokens[addr]?.unregister(); disconnectTokens[addr] = nil }
        emit(device, connected: false)
    }
    private func emit(_ d: IOBluetoothDevice, connected: Bool) {
        // device.name is UNTRUSTED (T-05-01) — passed as a plain String into the pure seam only.
        onReading(DeviceReading(name: d.name, classMajor: d.deviceClassMajor,
                                address: d.addressString, connected: connected))
    }

    func stop() {                                          // full teardown (Pitfall 5)
        connectToken?.unregister(); connectToken = nil
        disconnectTokens.values.forEach { $0.unregister() }; disconnectTokens.removeAll()
        running = false
    }
}
```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| paired Bluetooth device → app | A remote/paired device supplies `device.name` (an attacker-controllable String) and a device class across the IOBluetooth notification boundary. This is untrusted external input. |
| app process → IOBluetooth notification registrations | OS-held notification tokens (connect + per-device disconnect) are retained by the app; leaking/over-retaining them is a resource-integrity concern. |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-06-03 | Tampering / Info-disclosure | Untrusted `device.name` rendered in `deviceWings` (T-05-01) | mitigate | Passed as a plain `String` only into the already-tested pure `deviceLabel(...)` seam; never interpolated into a format string or shell; the SwiftUI `Text` is inert to format strings and is bounded with `.lineLimit(1) + .truncationMode(.tail)`. Verified by the acceptance-criteria grep for `truncationMode(.tail)`. |
| T-06-04 | Denial of Service (resource leak) | Orphaned IOBluetooth connect/disconnect tokens after the owner dies (T-03-06 analog) | mitigate | `BluetoothMonitor.stop()` unregisters the connect token AND every per-device disconnect token (`disconnectTokens.values.forEach { $0.unregister() }`); the controller's `deinit` (Plan 04) calls `stop()`, mirroring `PowerSourceMonitor`'s nonisolated teardown. |
| T-06-05 | Spoofing / Elevation (UX) | Speculative `NSBluetoothAlwaysUsageDescription` surfacing an unneeded TCC prompt | accept | Key deliberately NOT added (A1 verdict deferred to on-device). Adding it speculatively would surface a consent prompt where none is needed; the deferred on-device UAT settles whether it is required. No security risk from omission for the code-complete dry-run. |
| T-06-06 | Tampering | A future `-DDEBUG_BT_SPIKE` build double-registering IOBluetooth alongside the real monitor | mitigate | The entire spike + DEBUG_BT_SPIKE compile path is deleted (Task 3, verified by grep returning nothing), so no duplicate registration path can be compiled. |

No threat reaches **high** severity: the surface is a single local untrusted String (bounded + inert in SwiftUI Text) and OS notification tokens (fully torn down). The security gate passes.
