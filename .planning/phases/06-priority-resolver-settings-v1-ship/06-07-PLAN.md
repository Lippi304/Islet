<objective>
Fix five confirmed correctness bugs in the Phase-6 priority resolver / transient-queue controller
layer (`.planning/phases/06-priority-resolver-settings-v1-ship/06-PATTERNS.md` Bugs 1, 2, 3, and two
additional bugs the pattern-mapper did not cover but the live code confirms) surfaced by a fresh
multi-agent code review of the already-shipped Phase 6 code:

1. **Nil-address device-splash regression** (`handleDevice`) — a `guard let addr = reading.address
   else { return }` silently drops every addressless device reading BEFORE the pure
   `shouldShowDeviceSplash`/`deviceActivity(from:)` seam ever runs, contradicting that seam's own
   documented "nil address → can't dedup, but still show" contract (and the passing
   `testNilAddressConnectStillSplashes` unit test in `DeviceActivityTests.swift`).
2. **Battery-poll identity race** (`scheduleDeviceBatteryRefresh`) — the recursive poll matches the
   queue head by enum CASE SHAPE only, not by the address it was scheduled for, so a delayed poll for
   device A can overwrite device B's splash if B becomes head while A's poll is still in flight.
3. **Dismiss-timer not re-armed on promotion** (`flushTransients`) — when a category flush promotes a
   pending transient to head, the OLD dismiss timer (armed for the flushed transient's remaining
   window) keeps running unchanged instead of giving the newly-promoted transient a fresh ~3s window.
4. **Missed battery-refresh for promoted device** — `scheduleDeviceBatteryRefresh` is only called when
   a device becomes head IMMEDIATELY (`handleDevice`'s `if changed` branch); a device queued behind
   another transient and promoted later via `advance()` (or via a category flush) never gets its
   post-connect battery poll scheduled, so its wing can show a bare checkmark instead of the real %.
5. **`currentPresentation()` stale `isHealthy` flag** — when Now Playing is toggled off, `nowPlaying`
   is correctly forced to `.none` but `nowPlayingHealthy` is not, so a stale `false` from before the
   toggle can make `resolve(...)` render "nicht verfügbar" for a feature the user explicitly disabled.

Purpose: Restore the transient-queue's and device-battery-refresh's designed correctness guarantees
(D-02/D-03/D-04/D-09) without changing the pure `DeviceActivity`/`resolve(...)` seams' public
signatures — per the pattern-mapper's explicit guidance, identity tracking for the battery-poll race
belongs in a controller-owned side table (mirroring the existing `deviceLastShown: [String:
TimeInterval]` convention), NOT threaded into the `DeviceActivity` enum.
Output: Five targeted fixes in `NotchWindowController.swift`; one new pure helper in
`IslandResolver.swift` with a unit test in `IslandResolverTests.swift`.
</objective>

func shouldShowDeviceSplash(address: String?, connected: Bool, now: TimeInterval,
                            lastShown: [String: TimeInterval], debounce: TimeInterval,
                            suppressedAtLaunch: Set<String>) -> Bool {
    if let address {
        if connected && suppressedAtLaunch.contains(address) { return false }
        if let last = lastShown[address], now - last < debounce { return false }
    }
    return true   // nil address falls through to TRUE: "show it, just can't dedup/debounce it"
}

private func handleDevice(_ reading: DeviceReading) {
    let now = Date().timeIntervalSinceReferenceDate
    guard let addr = reading.address else { return }              // BUG (Finding 1): drops nil-address readings
    if reading.connected {
        guard !connectedDeviceAddresses.contains(addr) else { return }
        connectedDeviceAddresses.insert(addr)
        if let started = bluetoothStartedAt,
           Date().timeIntervalSince(started) < deviceLaunchGrace { return }
    } else {
        guard connectedDeviceAddresses.remove(addr) != nil else { return }
    }
    guard shouldShowDeviceSplash(address: addr, connected: reading.connected, now: now,
                                 lastShown: deviceLastShown, debounce: deviceDebounce,
                                 suppressedAtLaunch: deviceSuppressedAtLaunch)
    else { return }
    deviceLastShown[addr] = now
    guard let activity = deviceActivity(from: reading) else { return }
    deviceState.activity = activity
    let changed = transientQueue.enqueue(.device(activity))
    if changed {
        withAnimation(.spring(response: springResponse, dampingFraction: springDamping)) {
            renderPresentation()
        }
        updateVisibility()
        scheduleActivityDismiss()
        if reading.connected { scheduleDeviceBatteryRefresh(address: addr) }
    }
}

private func scheduleDeviceBatteryRefresh(address: String, attempt: Int = 0) {
    deviceBatteryWork?.cancel()
    guard attempt < 6 else { return }
    let work = DispatchWorkItem { [weak self] in
        guard let self else { return }
        guard case .device(.connected(let name, let glyph, let old))? = self.transientQueue.head
        else { return }                                            // BUG (Finding 2): category-only match, no identity
        if let monitor = self.bluetoothMonitor,
           let fresh = monitor.battery(forAddress: address), fresh != old {
            let updated = DeviceActivity.connected(name: name, glyph: glyph, battery: fresh)
            self.deviceState.activity = updated
            self.transientQueue.updateHead(.device(updated))
            withAnimation(.spring(response: self.springResponse, dampingFraction: self.springDamping)) {
                self.renderPresentation()
            }
        } else {
            self.scheduleDeviceBatteryRefresh(address: address, attempt: attempt + 1)
        }
    }
    deviceBatteryWork = work
    DispatchQueue.main.asyncAfter(deadline: .now() + 0.6, execute: work)
}

private func scheduleActivityDismiss() {
    dismissWorkItem?.cancel()
    let work = DispatchWorkItem { [weak self] in
        guard let self else { return }
        _ = self.transientQueue.advance()
        withAnimation(.spring(response: self.springResponse, dampingFraction: self.springDamping)) {
            self.syncActivityModels()
            self.renderPresentation()
        }
        self.updateVisibility()
        if self.transientQueue.head != nil {
            self.scheduleActivityDismiss()
        }
    }
    dismissWorkItem = work
    DispatchQueue.main.asyncAfter(deadline: .now() + activityDuration, execute: work)
}
